136 const string input(FILE *file, const string format, ...)
137 {
138  // print the args
139  if (format != NULL)
140  {
141  va_list vlis;
142  va_start(vlis, format);
143  vprintf(format, vlis);
144  va_end(vlis);
145  }
146 
147  // capacity of buffer
148  int capacity = 0;
149 
150  // size of buffer
151  int size = 0;
152 
153  // string buffer
154  string buffer = NULL;
155 
156  // current character
157  char c = 0;
158 
159  // read and apped character
160  while ((c = fgetc(file)) != '\n' && c != '\r' && c != EOF)
161  {
162  // is enough room
163  if (size + 1 > capacity)
164  {
165  // expand capacity
166  if (capacity < SIZE_MAX)
167  {
168  ++capacity;
169  }
170  else
171  {
172  free(buffer);
173  return NULL;
174  }
175 
176  // try to realloc
177  string temp = (string)realloc(buffer, capacity);
178 
179  // alloc fails
180  if (temp == NULL)
181  {
182  free(buffer);
183  return NULL;
184  }
185 
186  // assign temp
187  buffer = temp;
188  }
189 
190  // append char
191  buffer[size++] = c;
192  }
193 
194  // size is 0 the ""
195  if (size == 0)
196  {
197  return "";
198  }
199 
200  // if size is max then there is no room for NULL
201  if (size == SIZE_MAX)
202  {
203  free(buffer);
204  return NULL;
205  }
206 
207  //shrink buffer
208  string temp = (string)realloc(buffer, size + 1);
209 
210  // if fails then return NULL
211  if (temp == NULL)
212  {
213  free(buffer);
214  return NULL;
215  }
216  else
217  {
218  buffer = temp;
219  }
220 
221  // assign NULL
222  buffer[size] = '\0';
223 
224  // add to list
225  if (!lnlistadd(buffer))
226  {
227  free(buffer);
228  return NULL;
229  }
230 
231  // finally done
232  return buffer;
233 }
