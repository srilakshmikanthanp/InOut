139 const string read(FILE *file, const string format, ...)
140 {
141  // print the args
142  if (format != NULL)
143  {
144  va_list vlis;
145  va_start(vlis, format);
146  vprintf(format, vlis);
147  va_end(vlis);
148  }
149 
150  // capacity of buffer
151  int capacity = 0;
152 
153  // size of buffer
154  int size = 0;
155 
156  // string buffer
157  string buffer = NULL;
158 
159  // current character
160  char c = 0;
161 
162  // read and apped character
163  while ((c = fgetc(file)) != '\n' && c != '\r' && c != EOF)
164  {
165  // is enough room
166  if (size + 1 > capacity)
167  {
168  // expand capacity
169  if (capacity < SIZE_MAX)
170  {
171  ++capacity;
172  }
173  else
174  {
175  free(buffer);
176  return NULL;
177  }
178 
179  // try to realloc
180  string temp = (string)realloc(buffer, capacity);
181 
182  // alloc fails
183  if (temp == NULL)
184  {
185  free(buffer);
186  return NULL;
187  }
188 
189  // assign temp
190  buffer = temp;
191  }
192 
193  // append char
194  buffer[size++] = c;
195  }
196 
197  // size is 0 the ""
198  if (size == 0)
199  {
200  return "";
201  }
202 
203  // if size is max then there is no room for NULL
204  if (size == SIZE_MAX)
205  {
206  free(buffer);
207  return NULL;
208  }
209 
210  //shrink buffer
211  string temp = (string)realloc(buffer, size + 1);
212 
213  // if fails then return NULL
214  if (temp == NULL)
215  {
216  free(buffer);
217  return NULL;
218  }
219  else
220  {
221  buffer = temp;
222  }
223 
224  // assign NULL
225  buffer[size] = '\0';
226 
227  // add to list
228  if (!lnlistadd(buffer))
229  {
230  free(buffer);
231  return NULL;
232  }
233 
234  // finally done
235  return buffer;
236 }
